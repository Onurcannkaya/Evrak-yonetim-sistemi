    def restore_image_ultra(self, image: np.ndarray) -> np.ndarray:
        """
        ULTRA SEVİYE restorasyon - %80+ güven hedefi için.
        
        Bu metod en son çare olarak kullanılır. En yüksek kaliteli
        restorasyon için tüm teknikleri maksimum seviyede kullanır.

        İşlem Hattı:
            1. 4× Büyütme (LANCZOS4) - Maksimum detay
            2. Çift Gamma Düzeltme - Çok agresif parlaklık
            3. Süper güçlü CLAHE (clipLimit=16.0)
            4. Çoklu Unsharp Masking - Maksimum keskinlik
            5. Morphological Gradient - Kenar güçlendirme
            6. Top-hat transform - Arka plan temizleme
            7. Çok güçlü denoising
            8. Bilateral filter
            9. Sauvola threshold
            10. Akıllı morfolojik işlemler

        Args:
            image: BGR veya gri tonlama OpenCV görüntüsü.

        Returns:
            ULTRA seviyede restore edilmiş binary görüntü.
        """
        if image is None:
            raise ValueError("Geçersiz görüntü: None değeri alındı.")

        if len(image.shape) == 3:
            gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        else:
            gray = image.copy()

        # 1) 4× Upscale - LANCZOS4 (en kaliteli interpolasyon)
        upscaled = cv2.resize(
            gray, None, fx=4, fy=4, interpolation=cv2.INTER_LANCZOS4
        )

        # 2) Çift Gamma Düzeltme - Çok agresif
        # İlk geçiş: gamma=0.4 (çok parlak)
        gamma1 = 0.4
        inv_gamma1 = 1.0 / gamma1
        table1 = np.array([((i / 255.0) ** inv_gamma1) * 255
                          for i in np.arange(0, 256)]).astype("uint8")
        gamma_pass1 = cv2.LUT(upscaled, table1)
        
        # İkinci geçiş: gamma=0.6 (orta)
        gamma2 = 0.6
        inv_gamma2 = 1.0 / gamma2
        table2 = np.array([((i / 255.0) ** inv_gamma2) * 255
                          for i in np.arange(0, 256)]).astype("uint8")
        gamma_pass2 = cv2.LUT(gamma_pass1, table2)

        # 3) Süper Güçlü CLAHE - Maksimum kontrast
        clahe = cv2.createCLAHE(clipLimit=16.0, tileGridSize=(8, 8))
        enhanced = clahe.apply(gamma_pass2)

        # 4) Çoklu Unsharp Masking - Maksimum keskinlik
        # İlk geçiş
        gaussian1 = cv2.GaussianBlur(enhanced, (9, 9), 10.0)
        unsharp1 = cv2.addWeighted(enhanced, 3.0, gaussian1, -2.0, 0)
        
        # İkinci geçiş (daha ince detaylar)
        gaussian2 = cv2.GaussianBlur(unsharp1, (5, 5), 2.0)
        unsharp2 = cv2.addWeighted(unsharp1, 1.5, gaussian2, -0.5, 0)

        # 5) Morphological Gradient - Kenar güçlendirme
        kernel_grad = np.ones((3, 3), np.uint8)
        gradient = cv2.morphologyEx(unsharp2, cv2.MORPH_GRADIENT, kernel_grad)
        enhanced_edges = cv2.add(unsharp2, gradient)

        # 6) Top-hat Transform - Arka plan temizleme
        kernel_tophat = np.ones((15, 15), np.uint8)
        tophat = cv2.morphologyEx(enhanced_edges, cv2.MORPH_TOPHAT, kernel_tophat)
        enhanced_tophat = cv2.add(enhanced_edges, tophat)

        # 7) Çok Güçlü Denoising
        denoised = cv2.fastNlMeansDenoising(
            enhanced_tophat, None, h=30, templateWindowSize=7, searchWindowSize=21
        )

        # 8) Bilateral Filter - Kenarları koruyarak yumuşatma
        bilateral = cv2.bilateralFilter(
            denoised, d=15, sigmaColor=150, sigmaSpace=150
        )

        # 9) Sauvola-benzeri Adaptive Threshold
        binary = cv2.adaptiveThreshold(
            bilateral,
            255,
            cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY,
            blockSize=35,  # Daha büyük blok
            C=3,  # Daha hassas eşik
        )

        # 10) Akıllı Morfolojik İşlemler
        # Dilation - Harfleri kalınlaştır
        kernel_dilate = np.ones((2, 2), np.uint8)
        dilated = cv2.dilate(binary, kernel_dilate, iterations=1)
        
        # Closing - Harflerdeki boşlukları kapat
        kernel_close = np.ones((3, 3), np.uint8)
        closed = cv2.morphologyEx(dilated, cv2.MORPH_CLOSE, kernel_close)
        
        # Opening - Gürültüyü temizle
        kernel_open = np.ones((2, 2), np.uint8)
        cleaned = cv2.morphologyEx(closed, cv2.MORPH_OPEN, kernel_open)

        return cleaned
